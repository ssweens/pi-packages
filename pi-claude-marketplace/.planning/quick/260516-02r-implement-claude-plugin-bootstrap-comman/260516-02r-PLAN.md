<objective>
Add a `/claude:plugin bootstrap` subcommand that adds the
`anthropics/claude-plugins-official` marketplace to the **user scope** and
turns autoupdate **on** for that marketplace, atomically and idempotently.
Document the command in `README.md`.

`/claude:plugin bootstrap` to land in a working state with the canonical
Anthropic marketplace configured and tracking upstream automatically. Both
constituent operations (add, autoupdate) are already idempotent in the
codebase, so composing them satisfies the "do nothing if already set" rule
without any new state-mutation logic.

Output: New orchestrator + handler + router wire-up + tests + README block.

       ```typescript
       // edge/handlers/plugin/bootstrap.ts
       //
       // Quick 260516-02r: thin-shim handler factory for `/claude:plugin bootstrap`.
       // Delegates to `bootstrapClaudePlugin` orchestrator, threading deps.gitOps through.
       // Idempotent end-to-end -- both composed orchestrators are idempotent.

       import { bootstrapClaudePlugin } from "../../../orchestrators/plugin/bootstrap.ts";
       import { notifyError } from "../../../shared/notify.ts";
       import { parseCommandArgs } from "../../args-schema.ts";

       import type { ExtensionCommandContext } from "../../../platform/pi-api.ts";
       import type { EdgeDeps } from "../../types.ts";

       const USAGE = "Usage: /claude:plugin bootstrap";

       export function makeBootstrapHandler(
         deps: EdgeDeps,
       ): (args: string, ctx: ExtensionCommandContext) => Promise<void> {
         return async (args, ctx): Promise<void> => {
           // No positional args, no --scope flag accepted. parseCommandArgs
           // with `positional: []` still validates that no unknown flags slip
           // through; on any failure, notify usage and early-return.
           const parsed = parseCommandArgs(
             args,
             { positional: [] as const, usage: USAGE },
             (message) => { notifyError(ctx, message); },
           );
           if (parsed === undefined) {
             return;
           }

           // Reject --scope flag explicitly: bootstrap is user-scope only.
           if (parsed.scope !== undefined) {
             notifyError(ctx, `${USAGE}\n  bootstrap does not accept --scope; it always targets user scope.`);
             return;
           }

           await bootstrapClaudePlugin({
             ctx,
             cwd: ctx.cwd,
             gitOps: deps.gitOps,
           });
         };
       }
       ```
