```typescript
export interface InstallPluginOptions {
  readonly ctx: ExtensionContext;
  readonly pi?: ExtensionAPI;
  readonly scope: Scope;
  readonly cwd: string;
  readonly marketplace: string;  // PI-1 parse done in edge layer (Phase 6); orchestrator takes parsed pair
  readonly plugin: string;
}

/**
 * Local context type for the 5-phase ledger. Carries all state mutated
 * by phases. Not promoted to orchestrators/types.ts per D-01 corollary --
 * second-consumer rule, not in scope until/unless another orchestrator
 * needs the shape.
 */
interface InstallCtx {
  scope: Scope;
  locations: ScopedLocations;
  marketplace: string;
  plugin: string;
  resolved: ResolvedPluginInstallable;
  version: string;
  // Prep handles, populated by each phase.do before the phase commits.
  // Each phase.undo reads the handle and calls bridge unstage*.
  prepHandles: {
    skills?: PreparedSkills;
    commands?: PreparedCommands;
    agents?: PreparedAgents;
    mcp?: PreparedMcp;
  };
  // Generated names captured for soft-dep + reload-hint composition.
  stagedAgentNames: string[];
  stagedMcpServerNames: string[];
  stagedSkillNames: string[];
  stagedCommandNames: string[];
  // State snapshot, mutated in-place by the state-commit phase.
  stateSnapshot: ExtensionState;
}

export async function installPlugin(opts: InstallPluginOptions): Promise<void> {
  const { ctx, pi, scope, cwd, marketplace, plugin } = opts;
  const locations = locationsFor(scope, cwd);

  // Closure-capture bindings: declared BEFORE the try/withStateGuard so the
  // post-guard success path can compose soft-dep warnings + PI-13 dependencies
  // hint without re-entering the closure. The state-commit phase populates
  // these in-place as it builds installCtx.
  let stagedAgentNames: string[] = [];
  let stagedMcpServerNames: string[] = [];
  let resolvedNotes: readonly string[] = [];

  // Load fresh state inside the guard. PI-15 early sanity + PI-3/PI-4/PI-5
  // checks happen inside the closure so the eventual save fires atomically
  // on success and unwinds cleanly on throw (Phase 2 D-02).
  try {
    await withStateGuard(locations, async (state) => {
      const mp = state.marketplaces[marketplace];
      if (mp === undefined) {
        // PI-3 wording derives from PRD §5.2.1.
        throw new Error(`Plugin "${plugin}" not found in marketplace "${marketplace}".`);
      }
      // PI-5: already installed
      if (mp.plugins[plugin] !== undefined) {
        throw new Error(`Plugin "${plugin}" is already installed in marketplace "${marketplace}".`);
      }
      // Read cached manifest only (PI-2 -- NO network). manifestPath stored on mp record.
      const manifest = await loadCachedManifest(mp.manifestPath);  // helper -- read + validate via PLUGIN_MANIFEST_VALIDATOR
      const entry = manifest.plugins.find((p) => p.name === plugin);
      if (entry === undefined) {
        // PI-3 verbatim phrasing.
        throw new Error(`Plugin "${plugin}" not found in marketplace "${marketplace}".`);
      }

      // Resolve compatibility (Phase 2 D-04 strict resolver -- consumes Plan 05-03 array shape).
      const resolved = await resolveStrict(entry, { marketplaceRoot: mp.marketplaceRoot, /* fs hooks */ });
      // PI-4 wording derives from resolver.requireInstallable error message.
      const installable = requireInstallable(resolved, "install");

      // Compute generated names via domain/name.ts generators. These walk the
      // bridges' discover.ts to enumerate sources, NOT the on-disk staging
      // targets (no I/O against scope root yet).
      const discoveredSkills = await discoverPluginSkills({ pluginName: plugin, resolved: installable });
      const discoveredCommands = await discoverPluginCommands({ pluginName: plugin, resolved: installable });
      const discoveredAgents = await discoverPluginAgents({ pluginName: plugin, resolved: installable });

      const generatedNames = {
        skills:   discoveredSkills.map((s) => s.generatedName),
        commands: discoveredCommands.map((c) => c.generatedName),
        agents:   discoveredAgents.map((a) => a.generatedName),
      };

      // PI-6 / RN-3: pre-flight cross-bridge conflict guard. BEFORE any disk write.
      assertNoCrossPluginConflicts(scope, generatedNames, state);

      // PI-7 version: manifest > entry > hash
      const version = await resolveInstallVersion({
        manifestVersion: entry.version,
        installable,
      });
      // resolveInstallVersion -- private helper:
      //   if (manifestVersion !== undefined) return manifestVersion;
      //   if (entry has its own version field) return that;
      //   return await computeHashVersion(installable.pluginRoot);

      // Build the 5-phase ledger. D-01 literal-array discipline:
      // each phase has a SINGLE do/undo pair; the ctx is mutated in place.
      const installCtx: InstallCtx = {
        scope, locations, marketplace, plugin, resolved: installable, version,
        prepHandles: {},
        stagedAgentNames: [], stagedMcpServerNames: [],
        stagedSkillNames: [], stagedCommandNames: [],
        stateSnapshot: state,
      };

      const skillsPhase: Phase<InstallCtx> = {
        name: "skills",
        do: async (c) => {
          c.prepHandles.skills = await prepareStageSkills({ locations: c.locations, plugin: c.plugin, marketplace: c.marketplace, discovered: discoveredSkills });
          const committed = await commitPreparedSkills(c.prepHandles.skills);
          c.stagedSkillNames = committed.recorded.map((r) => r.generatedName);
        },
        undo: async (c) => {
          if (c.prepHandles.skills !== undefined) {
            await unstagePluginSkills({ locations: c.locations, plugin: c.plugin, marketplace: c.marketplace });
          }
        },
      };
      const commandsPhase: Phase<InstallCtx> = { /* same shape, commands bridge */ };
      const agentsPhase: Phase<InstallCtx> = { /* same shape, agents bridge */ };
      const mcpPhase: Phase<InstallCtx> = { /* same shape, mcp bridge */ };

      const statePhase: Phase<InstallCtx> = {
        name: "state",
        do: async (c) => {
          // Record the plugin in the snapshot. The guard saves on closure return.
          // PI-15 second check: re-read mp.plugins[plugin] under the guard's
          // closure-loaded state -- defensive given the guard's load-mutate-save
          // semantics; primarily protects against intra-process re-entry edge cases.
          if (c.stateSnapshot.marketplaces[c.marketplace]!.plugins[c.plugin] !== undefined) {
            throw new ConcurrentInstallError(c.plugin, c.marketplace);
          }
          c.stateSnapshot.marketplaces[c.marketplace]!.plugins[c.plugin] = {
            version: c.version,
            resolvedSource: c.resolved.source,
            compatibility: { installable: true },  // mirror the existing PluginRecord schema
            resources: {
              skills:     c.stagedSkillNames,
              prompts:    c.stagedCommandNames,
              agents:     c.stagedAgentNames,
              mcpServers: c.stagedMcpServerNames,
            },
            installedAt: new Date().toISOString(),
            updatedAt:   new Date().toISOString(),
          };
        },
        // undo: noop -- the guard has not yet flushed; on throw, the snapshot mutation
        // is discarded because the guard does not save when fn throws (ST-7 contract).
      };

      const phases: readonly Phase<InstallCtx>[] = [
        skillsPhase, commandsPhase, agentsPhase, mcpPhase, statePhase,
      ];

      const result = await runPhases(phases, installCtx);
      if (!result.ok) {
        // PI-14 bypass is inherited via Plan 05-02's formatRollbackError change.
        throw formatRollbackError(result, result.error!);
      }

      // Lift the post-guard-needed fields out of the closure into the
      // outer-scope let-bindings declared above the try.
      stagedAgentNames     = installCtx.stagedAgentNames;
      stagedMcpServerNames = installCtx.stagedMcpServerNames;
      resolvedNotes        = installable.notes ?? [];
    });
  } catch (err) {
    notifyError(ctx, formatErrorWithCauses(err as Error), err);
    return;
  }

  // POST-state-commit (AS-6): eager pluginDataDir mkdir.
  const dataDir = await locations.pluginDataDir(marketplace, plugin);
  try {
    await mkdir(dataDir, { recursive: true });
  } catch (err) {
    notifyWarning(
      ctx,
      `Plugin "${plugin}" installed; data dir creation deferred: ${errorMessage(err)}`,
    );
  }

  // Success notification. Soft-dep warnings (PI-11/PI-12) + reload hint (verb: load).
  // The closure has already populated stagedAgentNames / stagedMcpServerNames /
  // resolvedNotes above; read them directly.
  const subagentWarn = subagentWarningIfNeeded(pi, stagedAgentNames);
  const mcpWarn      = mcpAdapterWarningIfNeeded(pi, stagedMcpServerNames);
  let body = `Installed plugin "${plugin}" from marketplace "${marketplace}".`;
  if (subagentWarn !== "") body = `${body}\n${subagentWarn}`;
  if (mcpWarn !== "") body = `${body}\n${mcpWarn}`;

  // PI-13: dependencies declarations from resolved.notes -> manual-install warning.
  // The resolver appends `declares dependencies that must be installed manually`
  // to `notes` per PR-5 when the manifest entry carries a `dependencies` field.
  const depsNote = resolvedNotes.find((n) =>
    n.includes("dependencies that must be installed manually"),
  );
  if (depsNote !== undefined) {
    body = `${body}\n${depsNote}`;
  }

  const hint = reloadHint("load", [plugin]);
  notifySuccess(ctx, appendReloadHint(body, hint));
}
```

**Key implementation notes for the executor:**

1. **Closure capture pattern:** The success-message composition runs AFTER `withStateGuard` resolves. The simplest pattern is to declare `let stagedAgentNames: string[] = [];` (and siblings) BEFORE the `try { await withStateGuard(...) }` and have the closure populate them. After the guard returns, those let-bindings carry the data needed for `subagentWarningIfNeeded` etc.

2. **`loadCachedManifest` helper:** Read from `mp.manifestPath` (already validated and persisted by Phase 4 `marketplace add`/`update`); parse + validate via the manifest TypeBox validator. Throw on schema failure -- DOES NOT touch the network (PI-2).

3. **`resolveInstallVersion` helper:** Private. Pattern: `manifest entry .version ?? entry.version ?? await computeHashVersion(pluginRoot)`. Per CONTEXT.md "Open questions" researcher recommendation -- use plain `Error` with PI-3/PI-5 canonical messages (no new typed error class for these two paths).

4. **PI-15 detection:** Two layers per Pitfall 3 in PATTERNS.md -- (a) early-sanity check at TOP of guard closure to avoid running the ledger at all when the record already exists; (b) the state-commit phase re-asserts the record is still absent. Layer (a) is the common path; (b) protects against intra-process re-entry edge cases.

5. **PI-14:** Inherited via Plan 05-02's `formatRollbackError` chokepoint. The `catch (err)` here passes `err` straight to `notifyError` -- if it IS a PathContainmentError, formatRollbackError returned it verbatim (no rollback-partial wrapping); notify renders the original message.

6. **Architectural test (Plan 05-02):** `tests/architecture/no-orchestrator-network.test.ts` activates once this file lands. The file MUST NOT import `platform/git`, MUST NOT reference `DEFAULT_GIT_OPS`, MUST NOT have a `gitOps` field. Verify with grep before committing.
