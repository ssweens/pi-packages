// The implementation here is the 3-phase swap shared with updatePlugins.
// To preserve the cascade-safe contract, this top-level fn catches ALL
// throws and returns a partition='failed' outcome instead. PUP-9.

export const updateSinglePlugin: PluginUpdateFn = async (plugin, marketplace, scope) => {
  // Default cwd to process.cwd() since the cascade signature doesn't carry it;
  // Phase 7 wiring may override via a dependency-injection seam later.
  const cwd = process.cwd();
  try {
    return await runThreePhaseUpdate({
      plugin, marketplace, scope, cwd,
      ctx: undefined,            // cascade path: no user notifications during cascade
      pi: undefined,
      gitOps: DEFAULT_GIT_OPS,   // cascade owns its own gitOps lifecycle via the calling marketplace orchestrator
      cascade: true,             // routing flag for PUP-9
    });
  } catch (err) {
    // Cascade-safe: capture throws into partition='failed' outcome.
    return {
      partition: "failed",
      name: plugin,
      notes: [errorMessage(err)],
    };
  }
};

export async function updatePlugins(opts: UpdatePluginsOptions): Promise<void> {
  const { ctx, pi, cwd } = opts;
  const gitOps = opts.gitOps ?? DEFAULT_GIT_OPS;

  // Enumerate target (plugin, marketplace, scope) tuples per PUP-1.
  const targets = await enumerateTargets(opts);
  if (targets.length === 0) {
    // PUP-1 empty-set: silent success per PRD "No plugins installed."
    notifySuccess(ctx, "No plugins installed.");
    return;
  }

  // PUP-2 syncClone memoization -- per (scope, marketplace) pair.
  const synced = new Set<string>();
  async function syncCloneOnce(scope: Scope, mpName: string): Promise<void> {
    const key = `${scope}/${mpName}`;
    if (synced.has(key)) return;
    synced.add(key);
    const locations = locationsFor(scope, cwd);
    const state = await loadState(locations.extensionRoot);
    const mp = state.marketplaces[mpName];
    if (mp === undefined) throw new Error(`Marketplace "${mpName}" not found in ${scope} scope.`);
    // GitHub-source marketplace: fetch + forceUpdateRef + checkout (D-14 Phase 4 pattern).
    // For path-source marketplaces, syncClone is a noop (NFR-5: no network for path sources).
    if (mp.source.kind === "github") {
      const cloneDir = mp.marketplaceRoot;  // verify against actual record shape
      await gitOps.fetch(cloneDir);
      // ref + sha derivation: mirror orchestrators/marketplace/update.ts:393-467 logic
      // (the exact fetch+forceUpdateRef+checkout sequence)
      // For brevity here the executor should call into a shared helper if Phase 4
      // exposed one (refreshGitHubClone is a candidate); else inline the 3-step call.
    }
    // Re-read manifest after refresh -- but the partition enumeration below
    // does this per-plugin anyway, so syncCloneOnce only ensures the clone is fresh.
  }

  // PUP-2 partition + per-plugin update:
  const outcomes: PluginUpdateOutcome[] = [];
  for (const { plugin, marketplace, scope } of targets) {
    try {
      await syncCloneOnce(scope, marketplace);
    } catch (err) {
      // Pre-3-phase error -- direct path surfaces via notifyError (PUP-9).

  renderPartitionAndNotify(ctx, pi, outcomes);

// =============================================================================
// Shared 3-phase swap implementation
// =============================================================================

interface ThreePhaseArgs {
  plugin: string;
  marketplace: string;
  scope: Scope;
  cwd: string;
  ctx: ExtensionContext | undefined;     // undefined in cascade path
  pi: ExtensionAPI | undefined;
  gitOps: GitOps;
  cascade: boolean;
}

async function runThreePhaseUpdate(args: ThreePhaseArgs): Promise<PluginUpdateOutcome> {
  const { plugin, marketplace, scope, cwd, ctx, pi } = args;
  const locations = locationsFor(scope, cwd);

  // Pre-phase: resolve current vs new (PUP-3/4/5 short-circuits).
  const state = await loadState(locations.extensionRoot);
  const mp = state.marketplaces[marketplace];
  if (mp === undefined) return { partition: "skipped", name: plugin, notes: [`marketplace "${marketplace}" not found`] };
  const record = mp.plugins[plugin];
  if (record === undefined) return { partition: "skipped", name: plugin, notes: ["not installed"] };

  // Re-read refreshed manifest (synced by syncCloneOnce already).
  const manifest = await loadCachedManifest(mp.manifestPath);
  const entry = manifest.plugins.find((p) => p.name === plugin);
  if (entry === undefined) {
    // PUP-5
    return { partition: "skipped", name: plugin, notes: ["not in manifest"] };
  }

  // Resolver probe -- PUP-4 not-installable.
  let resolved;
  try {
    resolved = await resolveStrict(entry, { marketplaceRoot: mp.marketplaceRoot, /* fs hooks */ });
    requireInstallable(resolved, "update");   // throws "is no longer installable: ..." per PR-6
  } catch (err) {
    return { partition: "skipped", name: plugin, notes: [`no longer installable: ${errorMessage(err)}`] };
  }

  // PUP-3 unchanged: string equality (no semver) on version.
  const fromVersion = record.version;
  const toVersion = entry.version ?? fromVersion;  // version precedence echoes PI-7 if entry.version absent
  if (toVersion === fromVersion) {
    return { partition: "unchanged", name: plugin, fromVersion, toVersion };
  }

  // === Phase 1: prepare into tmp ===

  const discoveredSkills   = await discoverPluginSkills({ pluginName: plugin, resolved });
  const discoveredCommands = await discoverPluginCommands({ pluginName: plugin, resolved });
  const discoveredAgents   = await discoverPluginAgents({ pluginName: plugin, resolved });

  // PI-6 / RN-3 for update path: re-check cross-plugin conflicts on the
  // generated names for the NEW resolved plugin (names may have changed).
  const generatedNames = {
    skills:   discoveredSkills.map((s) => s.generatedName),
    commands: discoveredCommands.map((c) => c.generatedName),
    agents:   discoveredAgents.map((a) => a.generatedName),
  };
  // Exclude THIS plugin's currently-recorded resources from the comparison --
  // updating one's own names against one's own state is not a "cross-plugin"
  // conflict. The simplest approach: build a virtual state with this plugin's
  // record removed, then call assertNoCrossPluginConflicts.
  const stateForGuard = removePluginRecord(state, marketplace, plugin);
  assertNoCrossPluginConflicts(scope, generatedNames, stateForGuard);

  type PrepHandles = {
    skills?: Awaited<ReturnType<typeof prepareStageSkills>>;
    commands?: Awaited<ReturnType<typeof prepareStageCommands>>;
    agents?: Awaited<ReturnType<typeof prepareStagePluginAgents>>;
    mcp?: Awaited<ReturnType<typeof prepareStageMcpServers>>;
  };
  const prepHandles: PrepHandles = {};
  try {
    prepHandles.skills   = await prepareStageSkills({ locations, plugin, marketplace, discovered: discoveredSkills });
    prepHandles.commands = await prepareStageCommands({ locations, plugin, marketplace, discovered: discoveredCommands });
    prepHandles.agents   = await prepareStagePluginAgents({ locations, plugin, marketplace, discovered: discoveredAgents });
    prepHandles.mcp      = await prepareStageMcpServers({ locations, plugin, marketplace, /* mcpInput from resolved */ });
  } catch (err) {
    // Abort already-prepared handles (idempotent per Phase 3 D-04 corollary).
    // Pitfall 4: agents abort returns leak descriptor; mcp abort is sync void.
    const leaks: (string | undefined)[] = [];
    if (prepHandles.mcp !== undefined)      abortPreparedMcp(prepHandles.mcp);
    if (prepHandles.agents !== undefined)   leaks.push(await abortPreparedAgents(prepHandles.agents));
    if (prepHandles.commands !== undefined) await abortPreparedCommands(prepHandles.commands);
    if (prepHandles.skills !== undefined)   await abortPreparedSkills(prepHandles.skills);
    throw appendLeaks(err, leaks);
  }

  // === Phase 2: state-guard swap (with old-resource snapshot capture) ===

  let oldResources: typeof record.resources;
  try {
    await withStateGuard(locations, async (s) => {
      const sRecord = s.marketplaces[marketplace]?.plugins[plugin];
      if (sRecord === undefined) {
        // ST-9 concurrent change: plugin uninstalled between target enumeration and now.
        throw new Error(`Plugin "${plugin}" was concurrently uninstalled.`);
      }
      if (sRecord.version !== fromVersion) {
        // ST-9 concurrent change: another process updated this plugin.
        throw new Error(`Plugin "${plugin}" was concurrently updated; expected version "${fromVersion}", found "${sRecord.version}".`);
      }
      oldResources = sRecord.resources;
      sRecord.resources = {
        skills:     prepHandles.skills!.result.recorded.map((r) => r.generatedName),     // verify exact shape
        prompts:    prepHandles.commands!.result.recorded.map((r) => r.generatedName),
        agents:     prepHandles.agents!.result.recorded.map((r) => r.generatedName),
        mcpServers: prepHandles.mcp!.result.recorded.map((r) => r.generatedName),
      };
      sRecord.version = toVersion;
      sRecord.updatedAt = new Date().toISOString();
    });
  } catch (err) {
    // Phase 2 failure: abort all prep handles + rethrow.
    const leaks: (string | undefined)[] = [];
    if (prepHandles.mcp !== undefined)      abortPreparedMcp(prepHandles.mcp);
    if (prepHandles.agents !== undefined)   leaks.push(await abortPreparedAgents(prepHandles.agents));
    if (prepHandles.commands !== undefined) await abortPreparedCommands(prepHandles.commands);
    if (prepHandles.skills !== undefined)   await abortPreparedSkills(prepHandles.skills);
    throw appendLeaks(err, leaks);
  }

  // === Phase 3a: physical replace, aggregate failures across bridges ===

  const phase3aFailures: Phase3Failure[] = [];
  const commits: ReadonlyArray<readonly [Phase3Failure["phase"], () => Promise<void>]> = [
    ["skills",   () => commitPreparedSkills(prepHandles.skills!).then(() => undefined as unknown as void)],
    ["commands", () => commitPreparedCommands(prepHandles.commands!).then(() => undefined as unknown as void)],
    ["agents",   () => commitPreparedAgents(prepHandles.agents!).then(() => undefined as unknown as void)],
    ["mcp",      () => commitPreparedMcp(prepHandles.mcp!).then(() => undefined as unknown as void)],
  ];
  for (const [name, commit] of commits) {
    try {
      await commit();
    } catch (err) {
      phase3aFailures.push({ phase: name, msg: errorMessage(err), cause: err });
      // DO NOT fail-fast -- D-03 says continue across bridge failures so the
      // partial-replace state is fully observed.
    }
  }

  // === Phase 3b: error path with recovery hint, OR success ===

  if (phase3aFailures.length > 0) {
    // Compose the PUP-6 recovery hint via D-04 marker.
    const recoveryHint = `${RECOVERY_PLUGIN_REINSTALL_PREFIX} "${plugin}".`;
    const aggregateMsg =
      `Plugin "${plugin}" update failed during physical replace. ${recoveryHint}`;
    const aggregate = new PluginUpdatePhase3Error(aggregateMsg, phase3aFailures, {
      cause: phase3aFailures[0]?.cause,
    });
    // PUP-9 routing: direct path surfaces via notifyError; cascade path returns failed outcome.
    if (!args.cascade) {
      notifyError(args.ctx!, formatErrorWithCauses(aggregate), aggregate);
    }
    return { partition: "failed", name: plugin, fromVersion, toVersion, notes: [aggregateMsg] };
  }
