```typescript
// extensions/pi-claude-marketplace/orchestrators/plugin/uninstall.ts
//
// PU-1..8 + PU-7 propagation + AS-6 (post-commit cleanup leaks warning-severity).
//
// Composition (D-09):
//   withStateGuard(locations, async (state) => {
//     PU-5 silent converge: if record absent, set alreadyGone=true and return
//     outcome = await cascadeUnstagePlugin(plugin, marketplace, locations, installed)
//     if (!outcome.ok) throw outcome.cause -- PU-7 propagation
//     delete state.marketplaces[mp].plugins[plugin]
//     // guard saves on closure return
//   })
//   if (alreadyGone) return  -- PU-5 silent success
//   POST-state-commit: rm -rf pluginDataDir; leaks -> warning (PU-4)
//   PU-8 reload hint: only when ≥1 resource dropped

import { rm } from "node:fs/promises";

import { locationsFor } from "../../persistence/locations.ts";
import { appendReloadHint, reloadHint } from "../../presentation/reload-hint.ts";
import {
  mcpAdapterWarningIfNeeded,
  subagentWarningIfNeeded,
} from "../../presentation/soft-dep.ts";
import { appendLeaks, errorMessage } from "../../shared/errors.ts";
import { notifyError, notifySuccess, notifyWarning } from "../../shared/notify.ts";
import { withStateGuard } from "../../transaction/with-state-guard.ts";

import {
  cascadeUnstagePlugin,
  formatErrorWithCauses,
} from "../marketplace/shared.ts";

import type { UnstageOutcome } from "../marketplace/shared.ts";  // verify exact name
import type { ExtensionAPI, ExtensionContext } from "@mariozechner/pi-coding-agent";
import type { Scope } from "../../shared/types.ts";

export interface UninstallPluginOptions {
  readonly ctx: ExtensionContext;
  readonly pi?: ExtensionAPI;
  readonly scope: Scope;
  readonly cwd: string;
  readonly marketplace: string;
  readonly plugin: string;
}

export async function uninstallPlugin(opts: UninstallPluginOptions): Promise<void> {
  const { ctx, pi, scope, cwd, marketplace, plugin } = opts;
  const locations = locationsFor(scope, cwd);

  let alreadyGone = false;
  let outcome: UnstageOutcome | undefined;

  try {
    await withStateGuard(locations, async (state) => {
      const mp = state.marketplaces[marketplace];
      if (mp === undefined) {
        // Marketplace itself absent -- nothing to uninstall; treat as silent converge.
        alreadyGone = true;
        return;
      }
      const installed = mp.plugins[plugin];
      if (installed === undefined) {
        // PU-5 silent converge: record already gone (another process completed first
        // or there was never an install). PRD §5.2.2 specifies literal silence here.
        alreadyGone = true;
        return;
      }

      // PU-1 ordering enforced INSIDE cascadeUnstagePlugin (Phase 4 D-03 corollary).
      outcome = await cascadeUnstagePlugin(plugin, marketplace, locations, installed);

      // PU-7: cascade returns ok=false with chained AgentsUnstageFailureError when
      // foreign content detected at an agent target file. Re-throw to abort the
      // state commit (the marketplace record + plugin record stay intact for retry).
      if (!outcome.ok) {
        // outcome.cause is non-undefined when ok=false (Phase 4 D-03 contract).
        throw outcome.cause ?? new Error(`Cascade unstage failed for plugin "${plugin}".`);
      }

      // State commit: remove the plugin record. Guard saves atomically on closure return.
      // eslint-disable-next-line @typescript-eslint/no-dynamic-delete
      delete mp.plugins[plugin];
    });
  } catch (err) {
    notifyError(ctx, formatErrorWithCauses(err as Error), err);
    return;
  }

  // PU-5 silent converge: literal silence, no notification (CONTEXT.md Open
  // Questions researcher recommendation -- PRD §5.2.2 verbatim).
  if (alreadyGone) return;

  // POST-state-commit per PU-2 / D-08: drop the per-plugin data dir.
  // PU-4: cleanup leaks surface as warning-severity with the leaked path named.
  const dataDir = await locations.pluginDataDir(marketplace, plugin);
  const cleanupLeaks: string[] = [];
  try {
    await rm(dataDir, { recursive: true, force: true });
  } catch (err) {
    cleanupLeaks.push(`plugin data ${marketplace}/${plugin}: ${errorMessage(err)}`);
  }

  if (cleanupLeaks.length > 0) {
    notifyWarning(
      ctx,
      appendLeaks(new Error(`Plugin "${plugin}" removed; cleanup partial.`), cleanupLeaks).message,
    );
  }

  // PU-8 reload hint: only when ≥1 resource actually dropped.
  // `outcome` is defined here because alreadyGone is false (early-returned above).
  const droppedAny =
    outcome!.dropped.skills.length     > 0 ||
    outcome!.dropped.commands.length   > 0 ||
    outcome!.dropped.agents.length     > 0 ||
    outcome!.dropped.mcpServers.length > 0;

  // Soft-dep warnings for dropped agents/mcp when companion extensions unloaded.
  // RH-3..5 fire on drop too -- the resources won't actually unload until /reload.
  const subagentWarn = subagentWarningIfNeeded(pi, outcome!.dropped.agents);
  const mcpWarn      = mcpAdapterWarningIfNeeded(pi, outcome!.dropped.mcpServers);

  let body = `Uninstalled plugin "${plugin}" from marketplace "${marketplace}".`;
  if (subagentWarn !== "") body = `${body}\n${subagentWarn}`;
  if (mcpWarn !== "") body = `${body}\n${mcpWarn}`;

  const hint = reloadHint("drop", droppedAny ? [plugin] : []);
  notifySuccess(ctx, appendReloadHint(body, hint));
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Cascade output → state mutation | The cascade returns an `UnstageOutcome`; the orchestrator trusts `ok === false` implies `cause` is set (Phase 4 D-03 contract). Defensive fallback Error used if cause unexpectedly absent. |
| Agents-index → uninstall | Foreign content at agent target files is the PU-7 attack-surface concern. Detection lives in the agents bridge (Phase 3 D-06); orchestrator just re-throws to abort. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-5-02 | Tampering | Foreign agent content overwrite via cross-marketplace ownership | mitigate | PU-7: `cascadeUnstagePlugin` (Phase 4 D-03 + Phase 3 D-06) detects foreign content via the agent file's `pi-claude-marketplace-` prefix + `generated by pi-claude-marketplace` HTML-comment marker; the agents-index row is retained, the cascade returns `ok=false`, uninstall re-throws and the state record stays intact (retryable). |
| T-5-03 | EoP | Two processes uninstall the same plugin concurrently | mitigate | PU-5 silent converge at `withStateGuard` re-load: whichever process loses the race sees the record absent and exits silently. State.json atomic write (Phase 1 D-03) guarantees no half-written state. |
| T-5-04 | Information disclosure | Data dir cleanup failure exposes the leaked path | accept | PU-4 explicitly REQUIRES naming the leaked path in the warning body so the user can manually clean it up. The path is already under the scope root (controlled by user); there is no cross-tenant exposure surface. |
| T-5-17 | Tampering | Race between state-commit save and data-dir rm: process killed between → orphan data dir | accept | D-08 explicitly chose this ordering ("data dir AFTER state commit so EACCES cannot strand state in installed=true"). On retry, PU-5 silent converge applies; the orphan data dir is harmless (no longer referenced) and can be removed manually. |
