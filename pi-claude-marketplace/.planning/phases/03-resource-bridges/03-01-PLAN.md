    export function assertSafeName(name: string, label?: string): void {
      const ctx = label ? `${label} ` : "";
      if (typeof name !== "string") {
        throw new Error(`${ctx}name must be a string (got ${typeof name}).`);
      }
      if (name.trim() === "") {
        throw new Error(`${ctx}name must be a non-empty string.`);
      }
      if (name === "." || name === "..") {

      if (name.includes("/") || name.includes("\\")) {

      for (let i = 0; i < name.length; i++) {
        const code = name.charCodeAt(i);
        if (code < 0x20 || code === 0x7f) {
          throw new Error(`${ctx}name "${name}" must not contain ASCII control characters.`);
        }
      }
    }

    ## File 3: `extensions/pi-claude-marketplace/shared/fs-utils.ts`

    Two helpers: `cleanupStaging` (recursive rm) and `pathExists` (lstat-based predicate). `cleanupStaging` is ported from V1's `fs-utils.ts`. Recursive `rm` with `force: true` swallowing ENOENT. Returns `undefined` on success, descriptive leak message on failure (so callers can use `appendLeakToError(err, await cleanupStaging(...))` regardless). `pathExists` is added per B-06 fix so Plan 03-03's discover.ts can import it directly rather than inlining `lstat`.

    ```typescript
    import { lstat, rm } from "node:fs/promises";
    import { errorMessage } from "./errors.ts";

    /**
     * Best-effort recursive removal of a staging directory. Swallows ENOENT
     * (the dir was never created) and returns a descriptive leak message
     * for any other failure so the caller can surface it via
     * appendLeakToError without throwing from the cleanup itself.
     *
     * @returns `undefined` on success or ENOENT, a leak message string otherwise.
     */
    export async function cleanupStaging(dir: string, label: string): Promise<string | undefined> {
      try {
        await rm(dir, { recursive: true, force: true });
        return undefined;
      } catch (err) {
        const code = (err as NodeJS.ErrnoException).code;
        if (code === "ENOENT") return undefined;
        return `failed to clean up ${label} at ${dir}: ${errorMessage(err)}`;
      }
    }

    /**
     * lstat-based existence predicate. ENOENT/ENOTDIR -> false; any other
     * error propagates. Does NOT follow symlinks (consistent with PS-1).
     */
    export async function pathExists(p: string): Promise<boolean> {
      try {
        await lstat(p);
        return true;
      } catch (err) {
        const code = (err as NodeJS.ErrnoException).code;
        if (code === "ENOENT" || code === "ENOTDIR") return false;
        throw err;
      }
    }
    ```
