    ```typescript
    export interface ClaudeMarketplaceMarker {
      readonly plugin: string;
      readonly marketplace: string;
    }

    export function readMarker(value: unknown): ClaudeMarketplaceMarker | null {
      if (typeof value !== "object" || value === null || Array.isArray(value)) return null;
      const marker = (value as Record<string, unknown>)[CLAUDE_MARKETPLACE_MARKER_KEY];
      if (typeof marker !== "object" || marker === null || Array.isArray(marker)) return null;
      const obj = marker as Record<string, unknown>;
      if (typeof obj.plugin !== "string" || typeof obj.marketplace !== "string") return null;
      return { plugin: obj.plugin, marketplace: obj.marketplace };
    }

    export function buildMarker(plugin: string, marketplace: string): ClaudeMarketplaceMarker {
      return { plugin, marketplace };
    }

    export function isOwnedBy(value: unknown, plugin: string, marketplace: string): boolean {
      const marker = readMarker(value);
      return marker !== null && marker.plugin === plugin && marker.marketplace === marketplace;
    }
    ```

    ```typescript
    import { readFile } from "node:fs/promises";
    import path from "node:path";
    import { assertSafeName } from "../../domain/name.ts";
    import type { ResolvedMcpServers, ResolvePluginMcpServersInput } from "./types.ts";

    /**
     * MC-3 shape validation: must be an object, each entry must be an
     * object, each name must pass assertSafeName.
     */
    export function parseMcpServers(value: unknown, label: string): Record<string, unknown> {
      if (typeof value !== "object" || value === null || Array.isArray(value)) {
        throw new Error(`malformed ${label}: expected object, got ${Array.isArray(value) ? "array" : typeof value}`);
      }
      const obj = value as Record<string, unknown>;
      const out: Record<string, unknown> = {};
      for (const [name, entry] of Object.entries(obj)) {
        assertSafeName(name, `${label}.${name}`);
        if (typeof entry !== "object" || entry === null || Array.isArray(entry)) {
          throw new Error(`malformed ${label}.${name}: expected object`);
        }
        out[name] = entry;
      }
      return out;
    }

    /**
     * MC-1 precedence chain: marketplace-entry > plugin-manifest > standalone .mcp.json.
     * First match wins. Malformed at matched source THROWS -- does not fall through.
     */
    export async function resolvePluginMcpServers(input: ResolvePluginMcpServersInput): Promise<ResolvedMcpServers> {
      const { entry, manifest, pluginRoot } = input;

      if (entry.mcpServers !== undefined) {
        return {
          source: "marketplace-entry",
          servers: parseMcpServers(entry.mcpServers, "marketplace-entry mcpServers"),
        };
      }
      if (manifest.mcpServers !== undefined) {
        return {
          source: "plugin-manifest",
          servers: parseMcpServers(manifest.mcpServers, "plugin-manifest mcpServers"),
        };
      }

      const standalonePath = path.join(pluginRoot, ".mcp.json");
      let raw: string;
      try {
        raw = await readFile(standalonePath, "utf8");
      } catch (err) {
        const code = (err as NodeJS.ErrnoException).code;
        if (code === "ENOENT" || code === "ENOTDIR") return { source: "none", servers: {} };
        throw err;
      }

      let parsed: unknown;
      try {
        parsed = JSON.parse(raw);
      } catch (err) {
        throw new Error(`malformed standalone .mcp.json at ${standalonePath}: ${(err as Error).message}`);
      }

      // MC-2: detect wrapped vs unwrapped form
      let serversValue: unknown;
      if (
        typeof parsed === "object" && parsed !== null && !Array.isArray(parsed) &&
        "mcpServers" in (parsed as object)
      ) {
        serversValue = (parsed as { mcpServers: unknown }).mcpServers;
      } else {
        serversValue = parsed;
      }

      return {
        source: "standalone",
        servers: parseMcpServers(serversValue, `standalone ${standalonePath}`),
      };
    }
    ```

    ```typescript
    import { readFile } from "node:fs/promises";
    import { homedir } from "node:os";
    import path from "node:path";
    import { errorMessage } from "../../shared/errors.ts";

    /**
     * MC-4 / RN-5 user-contract slot list. The four pi-mcp-adapter
     * configuration paths checked for cross-slot collisions during stage.
     * Order is FIRST-DECLARER-WINS.
     */
    export function MCP_COLLISION_SLOTS(cwd: string): readonly string[] {
      return Object.freeze([
        path.join(homedir(), ".config", "mcp", "mcp.json"),        // shared-global
        path.join(homedir(), ".pi", "agent", "mcp.json"),          // pi-user-scope
        path.join(cwd, ".mcp.json"),                                // shared-project
        path.join(cwd, ".pi", "mcp.json"),                          // pi-project-scope
      ]);
    }

    /**
     * Walk all four pi-mcp-adapter slots and return Map<serverName, owningPath>.
     * First-declarer wins. Malformed JSON contributes nothing (silent skip
     * -- Pi runtime is what enforces those slots' validity).
     * EACCES propagates (not the bridge's concern).
     */
    export async function loadEffectiveServerNames(cwd: string): Promise<Map<string, string>> {
      const map = new Map<string, string>();
      for (const slotPath of MCP_COLLISION_SLOTS(cwd)) {
        let text: string;
        try {
          text = await readFile(slotPath, "utf8");
        } catch (err) {
          const code = (err as NodeJS.ErrnoException).code;
          if (code === "ENOENT" || code === "ENOTDIR") continue;
          throw err;
        }
        let parsed: unknown;
        try {
          parsed = JSON.parse(text);
        } catch {
          continue;  // malformed -- skip silently
        }
        // Each slot may be wrapped or unwrapped depending on the slot
        const servers = extractServers(parsed);
        for (const name of Object.keys(servers)) {
          if (!map.has(name)) map.set(name, slotPath);  // first-declarer wins
        }
      }
      return map;
    }

    function extractServers(parsed: unknown): Record<string, unknown> {
      if (typeof parsed !== "object" || parsed === null || Array.isArray(parsed)) return {};
      // Wrapped form: {mcpServers: {...}}
      if ("mcpServers" in (parsed as object)) {
        const inner = (parsed as { mcpServers: unknown }).mcpServers;
        if (typeof inner === "object" && inner !== null && !Array.isArray(inner)) {
          return inner as Record<string, unknown>;
        }
        return {};
      }
      // Unwrapped form: {server-name: {...}}
      return parsed as Record<string, unknown>;
    }
    ```

    ```typescript
    /**
     * MC-6 in-memory prepare: read scope's mcp.json, partition existing
     * entries by marker (ours vs theirs), MC-4 cross-slot collision check,
     * AS-8 noop short-circuit, MC-5 marker stamp on each new entry, build
     * merged doc.
     */
    export async function prepareStageMcpServers(input: StageMcpInput): Promise<PreparedMcpStaging> {
      const { locations, cwd, marketplaceName, pluginName, servers } = input;

      const doc = await readScopedDoc(locations.mcpJsonPath);
      const existing = getMcpServers(doc);

      const ours = new Set<string>();
      const theirs: Record<string, unknown> = {};
      for (const [name, value] of Object.entries(existing)) {
        if (isOwnedBy(value, pluginName, marketplaceName)) ours.add(name);
        else theirs[name] = value;
      }

      const newNames = Object.keys(servers);

      // MC-4 / RN-5 collision check across all 4 slots
      if (newNames.length > 0) {
        const effective = await loadEffectiveServerNames(cwd);
        for (const name of newNames) {
          if (ours.has(name)) continue;  // self-replace within own scope -- allowed

          const owningPath = effective.get(name);
          if (owningPath !== undefined && owningPath !== locations.mcpJsonPath) {
            throw new McpServerCollisionError(name, owningPath);
          }
          if (Object.prototype.hasOwnProperty.call(theirs, name)) {
            throw new McpServerCollisionError(name, locations.mcpJsonPath);
          }
        }
      }

      // AS-8 noop branch
      if (newNames.length === 0 && ours.size === 0) {
        return {
        kind: "noop",
        result: { stagedNames: [], recorded: [], warnings: [] },
      };
      }

      // MC-5 marker stamp
      const marker = buildMarker(pluginName, marketplaceName);
      const stamped: Record<string, unknown> = {};
      for (const [name, entry] of Object.entries(servers)) {
        const entryObj = (typeof entry === "object" && entry !== null && !Array.isArray(entry))
          ? (entry as Record<string, unknown>)
          : {};
        stamped[name] = { ...entryObj, [CLAUDE_MARKETPLACE_MARKER_KEY]: marker };
      }

      const next: RawMcpDoc = { ...doc, mcpServers: { ...theirs, ...stamped } };
      const recorded: StagedMcpRecord[] = newNames.map((generatedName) => ({
        generatedName,
        sourcePath: input.sourcePath ?? `${input.pluginName}#mcpServers`,
        targetPath: locations.mcpJsonPath,
      }));
      const result: StageMcpCommitResult = {
        stagedNames: Object.freeze(newNames),
        recorded: Object.freeze(recorded),
        warnings: [],
      };
      return { kind: "staged", locations, stagedNames: result.stagedNames, result, _nextDoc: next };
    }

    /** MC-6 commit: single atomicWriteJson; noop is zero-op. Returns StageMcpCommitResult for Phase 5 (W-05). */
    export async function commitPreparedMcp(prepared: PreparedMcpStaging): Promise<StageMcpCommitResult> {
      if (prepared.kind === "noop") return prepared.result;
      await atomicWriteJson(prepared.locations.mcpJsonPath, prepared._nextDoc);
      return prepared.result;
    }

    /** Abort: synchronous no-op (prepare wrote nothing to disk). */
    export function abortPreparedMcp(_prepared: PreparedMcpStaging): void {
      // no-op
    }
    ```

    ## File 2: `extensions/pi-claude-marketplace/bridges/mcp/unstage.ts`

    ```typescript
    import { readFile } from "node:fs/promises";

    import { atomicWriteJson } from "../../shared/atomic-json.ts";

    import { isOwnedBy } from "./marker.ts";
    import type { RawMcpDoc, UnstageMcpInput, UnstageMcpResult } from "./types.ts";

    /**
     * MC-7 unstage: read scope's mcp.json, drop entries with marker
     * matching (plugin, marketplace), atomic-write reduced doc.
     * Tolerates missing mcpServers field. Noop when nothing to remove.
     */
    export async function unstageMcpServers(input: UnstageMcpInput): Promise<UnstageMcpResult> {
      const { locations, marketplaceName, pluginName } = input;

      let doc: RawMcpDoc = {};
      try {
        const text = await readFile(locations.mcpJsonPath, "utf8");
        const parsed = JSON.parse(text);
        if (typeof parsed === "object" && parsed !== null && !Array.isArray(parsed)) {
          doc = parsed as RawMcpDoc;
        }
      } catch (err) {
        if ((err as NodeJS.ErrnoException).code === "ENOENT") {
          return { removedNames: Object.freeze([]), warnings: Object.freeze([]) };
        }
        // Malformed JSON we tolerate by treating as empty doc -- but MC-7 is
        // only about the mcpServers field being missing. Re-throw on parse fail.
        throw err;
      }

      const existing = (typeof doc.mcpServers === "object" && doc.mcpServers !== null && !Array.isArray(doc.mcpServers))
        ? (doc.mcpServers as Record<string, unknown>)
        : {};

      const removed: string[] = [];
      const kept: Record<string, unknown> = {};
      for (const [name, value] of Object.entries(existing)) {
        if (isOwnedBy(value, pluginName, marketplaceName)) removed.push(name);
        else kept[name] = value;
      }

      if (removed.length === 0) {
        return { removedNames: Object.freeze([]), warnings: Object.freeze([]) };
      }

      await atomicWriteJson(locations.mcpJsonPath, { ...doc, mcpServers: kept });
      return { removedNames: Object.freeze(removed), warnings: Object.freeze([]) };
    }
    ```
