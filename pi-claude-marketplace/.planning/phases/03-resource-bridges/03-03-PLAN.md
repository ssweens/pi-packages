    ```typescript
    import { readdir, lstat } from "node:fs/promises";
    import path from "node:path";

    import { assertSafeName, generatedSkillName } from "../../domain/name.ts";
    import { pathExists } from "../../shared/fs-utils.ts"; // or inline if not yet exported
    import type { ResolvedPluginInstallable } from "../../domain/resolver.ts";
    import type { DiscoveredSkill } from "./types.ts";

    export async function discoverPluginSkills(input: {
      pluginName: string;
      resolved: ResolvedPluginInstallable;
    }): Promise<readonly DiscoveredSkill[]> {
      // Phase 2 resolver guarantees componentPaths populated for installable plugins (W-03 fix:
      // dropped defensive ?? fallback; trust the Phase 2 contract).
      const skillsDir = input.resolved.componentPaths.skills;

      let entries: Awaited<ReturnType<typeof readdir>>;
      try {
        entries = await readdir(skillsDir, { withFileTypes: true });
      } catch (err) {
        const code = (err as NodeJS.ErrnoException).code;
        if (code === "ENOENT" || code === "ENOTDIR") return [];
        throw err;
      }

      const discovered: DiscoveredSkill[] = [];
      // Sort deterministically by name
      const sorted = [...entries].sort((a, b) => a.name.localeCompare(b.name));
      for (const entry of sorted) {
        // Skip dotfiles (e.g. ".gitkeep") -- planner-resolved hardening
        if (entry.name.startsWith(".")) continue;
        if (!entry.isDirectory()) continue;
        // Refuse symlinked entries (RESEARCH "Easy mistakes" #7)
        const full = path.join(skillsDir, entry.name);
        const stat = await lstat(full);
        if (stat.isSymbolicLink()) continue;  // skip symlinked skill dirs (defensive)

        assertSafeName(entry.name, `skill directory name in ${skillsDir}`);
        const skillMdPath = path.join(full, "SKILL.md");
        const skillStat = await lstat(skillMdPath).catch(() => null);
        if (!skillStat || !skillStat.isFile()) continue;
        if (skillStat.isSymbolicLink()) continue;  // refuse symlinked SKILL.md

        discovered.push({
          sourceName: entry.name,
          generatedName: generatedSkillName(input.pluginName, entry.name),
          skillDir: full,
        });
      }
      return discovered;
    }
    ```

    ```typescript
    /**
     * Rewrite SK-3 `name:` field in SKILL.md frontmatter, preserving all
     * other frontmatter. Adds frontmatter if absent.
     */
    export function rewriteFrontmatterName(content: string, newName: string): string {
      if (!content.startsWith("---")) {
        return `---\nname: ${newName}\n---\n\n${content}`;
      }
      const end = content.indexOf("\n---", 3);
      if (end === -1) {
        return `---\nname: ${newName}\n---\n\n${content}`;
      }
      const frontmatter = content.slice(3, end);
      const body = content.slice(end + 4);
      const nameRegex = /^name:.*$/m;
      let newFrontmatter: string;
      if (nameRegex.test(frontmatter)) {
        newFrontmatter = frontmatter.replace(nameRegex, `name: ${newName}`);
      } else {
        newFrontmatter = `\nname: ${newName}` + frontmatter;
      }
      return `---${newFrontmatter}\n---${body}`;
    }
    ```

    ```typescript
    import { mkdir, cp, readFile, writeFile, rename, rm } from "node:fs/promises";
    import { randomUUID } from "node:crypto";
    import path from "node:path";

    import { assertSafeName } from "../../domain/name.ts";
    import { assertPathInside } from "../../shared/path-safety.ts";
    import { substituteClaudeVars } from "../../shared/vars.ts";
    import { cleanupStaging } from "../../shared/fs-utils.ts";
    import { appendLeakToError } from "../../shared/errors.ts";

    import { discoverPluginSkills } from "./discover.ts";
    import { rewriteFrontmatterName } from "./rewrite-frontmatter.ts";
    import type {
      DiscoveredSkill,
      PreparedSkillsStaging,
      StageSkillsCommitResult,
      StageSkillsInput,
    } from "./types.ts";

    /**
     * RN-6 collision check: two source skill names that elide to the same
     * generated name MUST throw with both source names listed.
     */
    export function assertNoSkillCollisions(discovered: readonly DiscoveredSkill[]): void {
      const groups = new Map<string, string[]>();
      for (const s of discovered) {
        const arr = groups.get(s.generatedName) ?? [];
        arr.push(s.sourceName);
        groups.set(s.generatedName, arr);
      }
      const collisions: string[] = [];
      for (const [gen, sources] of groups) {
        if (sources.length > 1) {
          collisions.push(`"${gen}" <- [${sources.map((s) => `"${s}"`).join(", ")}]`);
        }
      }
      if (collisions.length > 0) {
        throw new Error(`Generated skill name collision detected. Rename one of the source skills:\n  ` + collisions.join("\n  "));
      }
    }

    export async function prepareStageSkills(input: StageSkillsInput): Promise<PreparedSkillsStaging> {
      const { locations, pluginName, pluginRoot, pluginDataDir, resolved } = input;
      const previousNames = input.previousSkillNames ?? [];
      const discovered = await discoverPluginSkills({ pluginName, resolved });
      assertNoSkillCollisions(discovered);

      // AS-8-style materialization gate (no REQ-ID for skills, but symmetry with mcp/agents)
      if (discovered.length === 0 && previousNames.length === 0) {
        return {
          kind: "noop",
          result: {
            stagedNames: Object.freeze([]),
            recorded: Object.freeze([]),
            warnings: Object.freeze([]),
          },
        };
      }

      const stagingRoot = path.join(locations.skillsStagingDir, randomUUID());
      await mkdir(stagingRoot, { recursive: true });
      await assertPathInside(locations.skillsStagingDir, stagingRoot, "skills staging root");

      const renamePairs: { from: string; to: string }[] = [];
      const stagedNames: string[] = [];

      try {
        for (const skill of discovered) {
          assertSafeName(skill.generatedName, "generated skill name");
          const stagedDir = path.join(stagingRoot, skill.generatedName);
          await assertPathInside(stagingRoot, stagedDir, "staged skill destination");

          const targetDir = path.join(locations.skillsTargetDir, skill.generatedName);
          await assertPathInside(locations.skillsTargetDir, targetDir, "skill target destination");

          // Copy source skill dir verbatim (preserve symlinks; do NOT dereference).
          await cp(skill.skillDir, stagedDir, {
            recursive: true,
            dereference: false,
            verbatimSymlinks: true,
            errorOnExist: true,
            force: false,
          });

          // SK-3 frontmatter name rewrite + SK-4 variable substitution on SKILL.md only.
          const skillMdPath = path.join(stagedDir, "SKILL.md");
          let content = await readFile(skillMdPath, "utf8");
          content = rewriteFrontmatterName(content, skill.generatedName);
          content = substituteClaudeVars(content, { pluginRoot, pluginData: pluginDataDir });
          await writeFile(skillMdPath, content, "utf8");

          renamePairs.push({ from: stagedDir, to: targetDir });
          stagedNames.push(skill.generatedName);
        }
      } catch (err) {
        throw appendLeakToError(err, await cleanupStaging(stagingRoot, "skills staging directory"));
      }

      const recorded: StagedSkillRecord[] = discovered.map((skill) => ({
        generatedName: skill.generatedName,
        sourcePath: skill.skillDir,
        targetPath: path.join(locations.skillsTargetDir, skill.generatedName),
      }));

      return {
        kind: "staged",
        locations,
        stagingRoot,
        result: {
          stagedNames: Object.freeze(stagedNames),
          recorded: Object.freeze(recorded),
          warnings: Object.freeze([]),
        },
        _previousNames: Object.freeze(previousNames),
        _renamePairs: Object.freeze(renamePairs),
      };
    }

    /**
     * Per-skill atomic dir rename. Removes previous-named target dirs first
     * (ENOENT-tolerant), then renames each staged dir into target. Cleans
     * up the staging UUID dir last (best-effort; failure surfaces but
     * does NOT undo the renames).
     */
    export async function commitPreparedSkills(prepared: PreparedSkillsStaging): Promise<string | undefined> {
      if (prepared.kind === "noop") return undefined;

      // Remove previous-named target dirs (re-stage path)
      for (const name of prepared._previousNames) {
        const dir = path.join(prepared.locations.skillsTargetDir, name);
        await assertPathInside(prepared.locations.skillsTargetDir, dir, "previous skill dir");
        try {
          await rm(dir, { recursive: true, force: true });
        } catch (err) {
          if ((err as NodeJS.ErrnoException).code !== "ENOENT") throw err;
        }
      }

      // mkdir target root + per-skill rename
      await mkdir(prepared.locations.skillsTargetDir, { recursive: true });
      for (const pair of prepared._renamePairs) {
        await rename(pair.from, pair.to);
      }

      // Best-effort cleanup of the staging UUID dir (the per-skill dirs were renamed away).
      return cleanupStaging(prepared.stagingRoot, "skills staging directory");
    }

    /**
     * abort: cleanup-only. Called when prepare returned "staged" but the
     * orchestrator decided not to commit. After commit succeeds, abort is a
     * no-op (staging is already cleaned).
     */
    export async function abortPreparedSkills(prepared: PreparedSkillsStaging): Promise<void> {
      if (prepared.kind === "noop") return;
      await cleanupStaging(prepared.stagingRoot, "skills staging directory");
    }
    ```

    ```typescript
    import { assertPathInside } from "../../shared/path-safety.ts";
    import type { UnstageSkillsInput, UnstageSkillsResult } from "./types.ts";

    export async function unstagePluginSkills(input: UnstageSkillsInput): Promise<UnstageSkillsResult> {
      const removed: string[] = [];
      for (const name of input.previousSkillNames) {
        const dir = path.join(input.locations.skillsTargetDir, name);
        await assertPathInside(input.locations.skillsTargetDir, dir, "skill to unstage");
        try {
          await rm(dir, { recursive: true, force: true });
          removed.push(name);
        } catch (err) {
          if ((err as NodeJS.ErrnoException).code !== "ENOENT") throw err;
        }
      }
      return { removedNames: Object.freeze(removed), warnings: Object.freeze([]) };
    }
    ```
