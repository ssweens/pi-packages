    ## Module

    ```typescript
    // extensions/pi-claude-marketplace/persistence/agents-index-io.ts
    import { readFile } from "node:fs/promises";

    import { atomicWriteJson } from "../shared/atomic-json.ts";
    import { errorMessage } from "../shared/errors.ts";

    import {
      AGENTS_INDEX_VALIDATOR,
      AGENTS_INDEX_ENTRY_VALIDATOR,
      type AgentsIndex,
      type AgentsIndexEntry,
    } from "./agents-index-schema.ts";
    import type { ScopedLocations } from "./locations.ts";

    /**
     * In-memory shape returned by loadAgentsIndex. Adds `corruptions` --
     * a frozen, per-row warning list the caller surfaces via warnings[]
     * in the bridge CommitResult.
     *
     * `corruptions` is NOT persisted. saveAgentsIndex accepts AgentsIndex,
     * not LoadedAgentsIndex.
     */
    export interface LoadedAgentsIndex {
      readonly schemaVersion: 1;
      readonly agents: readonly AgentsIndexEntry[];
      readonly corruptions: readonly string[];
    }

    /**
     * Load agents-index.json with AG-4 file-level-throw / per-row-drop split.
     *
     * Missing file → empty index (no throw).
     * Parse failure → throw.
     * schemaVersion missing or != 1 → throw.
     * agents field missing or not an array → throw.
     * Per-row validation failure → drop row, add to corruptions[].
     */
    export async function loadAgentsIndex(loc: ScopedLocations): Promise<LoadedAgentsIndex> {
      const indexPath = loc.agentsIndexPath;

      let text: string;
      try {
        text = await readFile(indexPath, "utf8");
      } catch (err) {
        if ((err as NodeJS.ErrnoException).code === "ENOENT") {
          return { schemaVersion: 1, agents: Object.freeze([]), corruptions: Object.freeze([]) };
        }
        throw err;
      }

      // File-level: parse
      let parsed: unknown;
      try {
        parsed = JSON.parse(text);
      } catch (err) {
        throw new Error(`Failed to parse agents-index at ${indexPath}: ${errorMessage(err)}`, { cause: err });
      }

      // File-level: schemaVersion check
      if (
        typeof parsed !== "object" ||
        parsed === null ||
        Array.isArray(parsed) ||
        (parsed as { schemaVersion?: unknown }).schemaVersion !== 1
      ) {
        throw new Error(`Unsupported agents-index schema at ${indexPath}: expected schemaVersion 1.`);
      }

      // File-level: agents-array check
      const obj = parsed as { schemaVersion: 1; agents?: unknown };
      if (!Array.isArray(obj.agents)) {
        throw new Error(`Invalid agents-index at ${indexPath}: 'agents' field must be an array.`);
      }

      // Per-row: validate each, accumulate corruptions
      const validAgents: AgentsIndexEntry[] = [];
      const corruptions: string[] = [];
      for (let i = 0; i < obj.agents.length; i++) {
        const row = obj.agents[i];
        if (AGENTS_INDEX_ENTRY_VALIDATOR.Check(row)) {
          validAgents.push(row as AgentsIndexEntry);
        } else {
          corruptions.push(`${indexPath}.agents[${i}]: row failed schema validation (entry dropped)`);
        }
      }

      return {
        schemaVersion: 1,
        agents: Object.freeze(validAgents),
        corruptions: Object.freeze(corruptions),
      };
    }

    /**
     * Save agents-index.json atomically. Validates the full document
     * against AGENTS_INDEX_VALIDATOR before write -- refuses on schema
     * violation rather than persisting bad data.
     */
    export async function saveAgentsIndex(loc: ScopedLocations, index: AgentsIndex): Promise<void> {
      if (!AGENTS_INDEX_VALIDATOR.Check(index)) {
        throw new Error(`saveAgentsIndex refused: index does not match AGENTS_INDEX_SCHEMA.`);
      }
      await atomicWriteJson(loc.agentsIndexPath, index);
    }
    ```

    Note on `loc.agentsIndexPath`: this field was added by Plan 03-01 Task 2. If executor finds it missing, the dependency on Plan 03-01 was violated -- investigate before proceeding.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| filesystem→loadAgentsIndex | Untrusted JSON content from `agents-index.json` crosses boundary; could be tampered, partial, or schema-evolved |
| caller→saveAgentsIndex | TypeScript-typed `AgentsIndex` value crosses; runtime validator re-checks before disk write |
| filesystem→atomicWriteJson | TOCTOU window between schema validation and disk write |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-08 | Tampering (untrusted JSON) | loadAgentsIndex | mitigate | Schema validation BEFORE any field access. File-level checks throw immediately. Per-row checks drop+warn. AG-4 discipline. |
| T-03-09 | Information Disclosure (path leak in errors) | loadAgentsIndex | accept | Error messages include `indexPath` and `agents[N]` indices. Local-only paths; not exposed beyond CLI. |
| T-03-10 | Denial of Service (huge agents-index.json) | loadAgentsIndex | accept | TypeBox JIT validator is O(n) over rows; an attacker writing a 100MB file would impact only the local process. Max file size is a Phase 7 concern (NFR-8 BACKLOG mtime caching). |
| T-03-11 | Elevation of Privilege (path traversal via agentsIndexPath) | saveAgentsIndex | mitigate | `loc.agentsIndexPath` is constructed at `locationsFor` time from `extensionRoot` + a HARDCODED literal `"agents-index.json"`. No untrusted name component reaches the path. |
| T-03-12 | Tampering (TOCTOU between validate and write) | saveAgentsIndex | accept | `atomicWriteJson` writes to tmp + rename -- concurrent writers serialized by write-file-atomic's queue. The validator runs in-memory before the disk call; a concurrent process replacing the file between validation and atomicWriteJson cannot corrupt the saved value (rename is atomic). |
| T-03-13 | Repudiation (silent row drop) | loadAgentsIndex | mitigate | Per-row drops surface in `corruptions[]` so callers can route through `notifyWarning` (IL-3 sanctioned `console.warn` for the load path). The bridge unit tests assert `corruptions.length` is non-empty for malformed fixtures. |
